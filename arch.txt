啟動服務：
sudo sed -i 's/deny = 3/deny = 0/' /etc/security/faillock.conf

/etc/security/faillock.conf的deny=0取消註解

下載yay
git clone https://aur.archlinux.org/yay.git ~/build/yay
cd ~/build/yay
makepkg -si

fc-cache -fv

yay -S google-chrome
yay -S tty-clock

sudo systemctl enable --now libvirtd
sudo systemctl enable --now virtnetworkd
sudo systemctl enable --now virtstoraged
sudo systemctl enable --now tailscaled
sudo systemctl enable --now docker
sudo systemctl enable --now ufw

sudo virsh net-start default
sudo virsh net-autostart default
sudo usermod -aG libvirt,kvm $USER
sudo usermod -aG docker $USER
sudo tailscale up
sudo tailscale up --ssh

/etc/environment：
# 輸入法 (你原本的)
XMODIFIERS=@im=fcitx
SDL_IM_MODULE=fcitx

# 讓 Chrome 支援 Wayland 且原生跑在 GPU 上
ELECTRON_OZONE_PLATFORM_HINT=auto

# --- 以下為 NVIDIA 關鍵補強 ---
# 強制使用 NVIDIA 的 GBM 後端 (Niri 必備)
GBM_BACKEND=nvidia-drm
# 讓應用程式介面由 NVIDIA 負責渲染
__GLX_VENDOR_LIBRARY_NAME=nvidia
# 解決滑鼠游標在某些 Wayland 下不見的問題
WLR_NO_HARDWARE_CURSORS=1
# 修正特定工具的 Vulkan 支援
ENABLE_VULKAN_RENDERER=1


/etc/mkinitcpio.conf：
MODULES=(i915 nvidia nvidia_modeset nvidia_uvm nvidia_drm)
（終端機輸入sudo mkinitcpio -P)
